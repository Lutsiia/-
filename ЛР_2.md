### Текст программы

#include <iostream>
using namespace std;

int main ()
{
    setlocale(LC_ALL, "Russian");
    int N1, M1, A[10][10];
    int N2, M2, B[10][10];
    int C[10][10];
    cout << "Введите количество строк матрицы А: ";
    cin >> N1;
    cout << "Введите количество столбцов матрицы А: ";
    cin >> M1;
    cout << "Введите количество строк матрицы В: ";
    cin >> N2;
    cout << "Введите количество столбцов матрицы В: ";
    cin >> M2;
    if (M1 != N2)
    {
        cout << "Умножение невозможно\n";
        return 0;
    }
    
    for (int i = 0; i < N1; i++){
        for (int j = 0; j < M1; j++){
            cout << " A[" << i + 1 << "," << j + 1 << "]=";
            cin >> A[i][j];
            }
    }
    
    for (int i = 0; i < N1; i++){
        for (int j = 0; j < M1; j++){
            cout << A[i][j] << "\t";
        }
        cout << endl;
    }
    
    for (int i = 0; i < N2; i++){
        for (int j = 0; j < M2; j++){
            cout << "B[" << i + 1 << "]=";
            cin>> B[i][j];
        }
    }
    
    for (int i = 0; i < N2; i++){
        for (int j = 0; j < M2; j++){
            cout << B[i][j] << "\t";
        }
        cout << endl;
    }
    
    for ( int i=0; i<N1; i++){
        for (int j=0; j<M1; j++){
            C[i][j] = 0;
            for (int l = 0; l < M1; l++){
                C[i][j] += A[i][l] * B[l][j];
                
            }
        }
    }
    
    cout << "C = " << endl;
    for (int i=0; i<N1; i++){
        for (int j = 0; j<M2; j++){
            cout << C[i][j] << "\t";
            
        }
        cout << endl;
    }
    system("pause");
    return 0;
}
